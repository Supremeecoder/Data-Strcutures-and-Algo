#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(string s,int num) {
    char a=s[0];
    int ans=0;
    int count[26]={0};
    int count1[26]={0};
    for(int i=1;i<num;i++) {
        if(a==s[i]) {
            ans=i;
            break;
        }
    }
    if(ans==(num-1)) {
        return num;
    }
    for(int i=0;i<ans;i++) {
        count[s[i]-'a']++;
    }
    for(int i=ans;i<num;i++) {
        count1[s[i]-'a']++;
    }
    for(int i=ans;i<num;i++) {
        cout<<count1[i]<<" ";
    }
    cout<<endl;
    for(int i=ans;i<num;i++) {
        cout<<count1[i]<<" ";
    }
    cout<<endl;
    int len;
    int len1;
    for(int i=0;i<26;i++) {
        if(count[i]>=1) {
            len++;
        }
    }
    for(int i=0;i<26;i++) {
        if(count1[i]>=1) {
            len1++;
        }
    }
    return len+len1;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
    cin>>num;
    string s;
    getline(cin,s);
    cout<<findans(s,num);
    }
    return 0;
}
