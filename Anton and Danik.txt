#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(string s,int num) {
    int counta=0;
    int countd=0;
    for(int i=0;i<s.size();i++) {
        if(s[i]=='A') {
            counta++;
        } else if(s[i]=='D') {
            countd++;
        }
    }
    if(counta>countd) {
        cout<<"Anton";
    } else if(countd>counta) {
        cout<<"Danik";
    } else {
        cout<<"Friendship";
    }
}
int main()
{
    string s;
    int num;
    cin>>num;
    cin>>s;
    findans(s,num);
    return 0;
}
