#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int num1,int num2,int num3) {
    int res=num3-num2;
    return (res/num1)*num1+num2;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num1;
        cin>>num1;
        int num2;
        cin>>num2;
        int num3;
        cin>>num3;
        cout<<findans(num1,num2,num3);
        cout<<endl;
    }
    return 0;
}
