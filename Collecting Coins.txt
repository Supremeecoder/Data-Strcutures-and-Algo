#include<iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int num1,int num2,int num3,int num) {
    int res=max(num1,num2);
    int res1=max(res,num3);
    int count=res1-num1+res1-num2+res1-num3;
    if((num-count)<0) {
        cout<<"NO"<<endl;
        return;
    }
    if(((num-count)%3)==0) {
        cout<<"YES"<<endl;
        return;
    }
    cout<<"NO"<<endl;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num1;
        int num2;
        int num3;
        int num;
        cin>>num1>>num2>>num3>>num;
        findans(num1,num2,num3,num);
    }
    return 0;
}
