#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void ans() {
    int n;
    cin>>n;
    vector<string> a(2*(n-1));
    for(int i=0;i<2*(n-1);i++) {
        cin>>a[i];
    }
    vector<string> s;
    for(int i=0;i<2*(n-1);i++) {
        if(a[i].size()==n-1) {
            s.push_back(a[i]);
        }
    }
    if(s[0].substr(1)!=s[1].substr(0,n-2)) {
        swap(s[0],s[1]);
    }
    string t=s[0]+s[1][n-2];
    string t1=t;
    reverse(t1.begin(),t1.end());
    if(t==t1) {
        cout<<"YES"<<endl;
    } else {
        cout<<"NO"<<endl;
    }
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        ans();
    }
    return 0;
}
