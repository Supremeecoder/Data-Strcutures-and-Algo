#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int arr[],int num,int h) {
    int count=0;
    for(int i=0;i<num;i++) {
        if(arr[i]>h) {
            count++;
        }
        count++;
    }
    return count;
}
int main()
{
    int num;
    int h;
    cin>>num;
    cin>>h;
    int arr[num];
    for(int i=0;i<num;i++) {
        cin>>arr[i];
    }
    cout<<findans(arr,num,h);
    return 0;
}
