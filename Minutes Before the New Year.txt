#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int num1,int num2) {
    return (60*(24-num1-1)+(60-num2));
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num1,num2;
        cin>>num1>>num2;
        cout<<findans(num1,num2);
        cout<<endl;
    }
    return 0;
}
