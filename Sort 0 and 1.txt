#include <iostream>
using namespace std;
void sort0n1(int arr[],int num) {
    int low=0;
    int high=num-1;
    while(low<high) {
        if(arr[low]==1) {
            if(arr[high]==0) {
                swap(arr[low],arr[high]);
            }
            high--;
        } else {
            low++;
        }
    }
    for(int i=0;i<num;i++) {
        cout<<arr[i]<<" ";
    }
}
int main()
{
    int arr[]={1,0,0,1,0,1,0,1,1,0};
    int len=sizeof(arr)/sizeof(arr[0]);
    sort0n1(arr,len);
    return 0;
}
