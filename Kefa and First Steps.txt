#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int num,int arr[]) {
    if(num==1 ||num==0) {
       return num;
    }
    int count=1;
    int res=1;
    for(int i=0;i<num-1;i++) {
        if(arr[i]<=arr[i+1]) {
            count++;
            res=max(res,count);
        } else {
            count=1;
        }
    }
    return res;
}
int main()
{
    int num;
    cin>>num;
    int arr[num];
    for(int i=0;i<num;i++) {
        cin>>arr[i];
    }
    cout<<findans(num,arr);
    return 0;
}
