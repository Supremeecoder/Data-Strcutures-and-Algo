#include<bits/stdc++.h>
using namespace std;
void findans(string s1) {
    unordered_set<char> us;
    for(int i=0;i<s1.size();i++) {
        us.insert(s1[i]);
    }
    int len=us.size();
    if(len%2==0) {
        cout<<"CHAT WITH HER!";
    } else {
        cout<<"IGNORE HIM!";
    }
    return;
}
