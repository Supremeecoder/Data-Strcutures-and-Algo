#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int arr[],int num,int k) {
    int count=0;
    for(int i=0;i<num;i++) {
        if(arr[i]>0) {
            if(arr[i]>=arr[k-1]) {
            count++;
        }
        }
    }
    return count;
}
int main()
{
        int num;
        int k;
        cin>>num;
        cin>>k;
        int arr[num];
        for(int i=0;i<num;i++) {
            cin>>arr[i];
        }
        cout<<findans(arr,num,k);
    return 0;
}
