#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int num,string s,int k) {
    int arr1[26]={0};
    int arr2[26]={0};
    int arr3[26]={0};
    for(int i=0;i<num;i++) {
        if(islower(s[i])) {
            arr1[s[i]-'a']++;
        } else {
            arr2[s[i]-'A']++;
        }
    }
    // for(int i=0;i<26;i++) {
    //     cout<<arr1[i]<<" ";
    // }
    // cout<<endl;
    // for(int i=0;i<26;i++) {
    //     cout<<arr2[i]<<" ";
    // }
    // cout<<endl;
    int count=0;
    for(int i=0;i<26;i++) {
        count=count+min(arr1[i],arr2[i]);
        arr3[i]=abs(arr1[i]-arr2[i]);
    }
    // for(int i=0;i<26;i++) {
    //     cout<<arr3[i]<<" ";
    // }
    // cout<<endl;
    // cout<<count<<endl;
    int i=0;
    while(k>0 && i<26) {
        if(arr3[i]>1) {
            if(k-(arr3[i]/2)<0)
            {
                count+=k;
                break;
            }
            else
            {
            k=k-arr3[i]/2;
            count=count+arr3[i]/2;
            }
        }
        i++;
    }
    return count;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        int k;
        cin>>k;
        string s;
        cin>>s;
        cout<<findans(num,s,k);
        cout<<endl;
    }
    return 0;
}
