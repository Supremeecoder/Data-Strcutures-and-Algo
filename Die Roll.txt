#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int num,int num1) {
    int res=max(num,num1);
    if(res>6) {
      cout<<"0/1"<<endl;
      return;
    }
    if(res==6) {
      cout<<"1/6"<<endl;
      return;
    }if(res==5) {
      cout<<"1/3"<<endl;
      return;
    }if(res==4) {
      cout<<"1/2"<<endl;
      return;
    }if(res==3) {
      cout<<"2/3"<<endl;
      return;
    }
    if(res==2) {
      cout<<"5/6"<<endl;
      return;
    }if(res==1) {
      cout<<"1/1"<<endl;
      return;
    }
}
int main()
{
    
    int num1;
    int num2;
    cin>>num1>>num2;
    findans(num1,num2);
    return 0;
}
