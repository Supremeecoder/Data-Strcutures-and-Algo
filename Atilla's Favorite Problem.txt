#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(string s,int num) {
    char res=s[0];
    for(int i=1;i<s.size();i++) {
        if(res<s[i]) {
            res=s[i];
        }
    }
    return res-'a'+1;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        string s;
        cin>>s;
        cout<<findans(s,num);
        cout<<endl;
    }
    return 0;
}
