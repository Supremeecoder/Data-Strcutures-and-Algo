#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(string s) {
    if(s[0]=='T') {
        return 4;
    }
    if(s[0]=='C') {
        return 6;
    }
    if(s[0]=='O') {
        return 8;
    }
    if(s[0]=='D') {
        return 12;
    }
    if(s[0]=='I') {
        return 20;
    }
}
int main()
{
    int num;
    cin>>num;
    int count=0;
    while(num--) {
        string s;
        cin>>s;
        count=count+findans(s);
    }
    cout<<count;
    return 0;
}
