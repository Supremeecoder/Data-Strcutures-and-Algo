#include <iostream>
#include<bits/stdc++.h>
using namespace std;
string findans(string s) {
    string str="";
    int count=0;
    for(int i=0;i<s.size();i++) {
        if(s[i]=='W'&&s[i+1]=='U'&&s[i+2]=='B') {
            i+=2;
            continue;
        } else {
            if(s[i-1]=='B' && count>=1 && s[i-2]=='U' && s[i-3]=='W') {
                str=str+" ";
            }
            count++;
            str=str+s[i];
        }
    }
    return str;
}
int main()
{
    string s;
    cin>>s;
    cout<<findans(s);
    return 0;
}
