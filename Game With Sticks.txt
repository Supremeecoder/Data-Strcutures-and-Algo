#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int num,int num1) {
    int res=min(num,num1);
    if(res%2==0) {
        cout<<"Malvika"<<endl;
        return;
    }
    cout<<"Akshat"<<endl;
}
int main()
{
    int num1;
    int num2;
    cin>>num1>>num2;
    findans(num1,num2);
    return 0;
}
