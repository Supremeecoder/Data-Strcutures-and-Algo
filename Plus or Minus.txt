#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int num1,int num2,int num3) {
    if((num1+num2)==num3) {
        cout<<"+"<<endl;
        return;
    } else if((num1-num2)==num3) {
        cout<<"-"<<endl;
        return;
    }
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num1;
        int num2;
        int num3;
        cin>>num1>>num2>>num3;
        findans(num1,num2,num3);
    }
    return 0;
}
