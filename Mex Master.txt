#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int arr[],int num) {
    int count0=0;
    for(int i=0;i<num;i++) {
        if(arr[i]==0) {
            count0++;
        }
    }
    if(count0<=(num+1)/2) {
        return 0;
    } else if(arr[num-1]!=1) {
        return 1;
    } else {
        return 2;
    }
    return 0;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        int arr[num];
        for(int i=0;i<num;i++) {
            cin>>arr[i];
        }
        cout<<findans(arr,num);
        cout<<endl;
    }
    return 0;
}
