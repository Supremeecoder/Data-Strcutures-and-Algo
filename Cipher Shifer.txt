#include <iostream>
using namespace std;
void findans(int len,string s) {
    string ans="";
    char st=s[0];
    for(int i=1;i<len;i++) {
        if(s[i]==st) {
            ans=ans+s[i];
            st=s[i+1];
            i++;
        }
    }
    cout<<ans<<endl;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int len;
    cin>>len;
    string s;
    cin>>s;
    findans(len,s);
    }
    return 0;
}
