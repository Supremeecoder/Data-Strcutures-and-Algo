#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int num1,int num2) {
    int count=0;
    while(num1<=num2) {
        num1=num1*3;
        num2=num2*2;
        count++;
    }
    return count;
}
int main()
{
    int num;
    int num1;
    cin>>num;
    cin>>num1;
    cout<<findans(num,num1);
    return 0;
}
