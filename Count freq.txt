#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void countfreq(int arr[],int num) {
    unordered_map<int,int> mp;
    for(int i=0;i<num;i++) {
        mp[arr[i]]++;
    }
    for(auto x:mp) {
        cout<<x.first<<" "<<x.second<<endl;
    }
}
int main()
{
    int arr[]={10,20,40,60,10,10,10,20,30,40};
    int len=sizeof(arr)/sizeof(arr[0]);
    countfreq(arr,len);
    return 0;
}
