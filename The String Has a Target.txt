#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int num,string s) {
    char min=s[0];
    int temp;
    for(int i=0;i<s.size();i++) {
        if(min>=s[i]) {
            min=s[i];
            temp=i;
        }
    }
    cout<<min;
    for(int i=0;i<temp;i++) {
        cout<<s[i];
    }
    for(int i=temp+1;i<s.size();i++) {
        cout<<s[i];
    }
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        string s;
        cin>>s;
        findans(num,s);
        cout<<endl;
    }
    return 0;
}
