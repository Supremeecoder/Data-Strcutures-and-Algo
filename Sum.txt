#include <iostream>
using namespace std;
void findans(int num1,int num2,int num3) {
    if((num1+num2)==num3 ||(num1+num3)==num2 ||(num3+num2)==num1) {
        cout<<"YES"<<endl;
        return;
    }
    cout<<"NO"<<endl;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num1,num2,num3;
        cin>>num1>>num2>>num3;
        findans(num1,num2,num3);
    }
    return 0;
}
