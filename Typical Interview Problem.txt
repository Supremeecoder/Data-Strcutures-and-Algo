#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(string s,int num) {
    string str="";
    int x=1;
    while(str.size()<100) {
        if(x%3==0) {
            str=str+'F';
        }
        if(x%5==0) {
            str=str+'B';
        }
        x++;
    }
    if(str.find(s)!=-1) {
        cout<<"YES"<<endl;
        return;
    }
    cout<<"NO"<<endl;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        string s;
        cin>>num;
        cin>>s;
        findans(s,num);
    }
    return 0;
}
