#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int num,int arr[]) {
    sort(arr,arr+num);
    for(int i=0;i<num;i++) {
        cout<<arr[i]<<" ";
    }
}
int main()
{
    int num;
    cin>>num;
    int arr[num];
    for(int i=0;i<num;i++) {
        cin>>arr[i];
    }
    findans(num,arr);
    return 0;
}
