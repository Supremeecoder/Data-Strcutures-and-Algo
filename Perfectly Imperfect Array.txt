#include <iostream>
#include <bits/stdc++.h>

using namespace std;
int main(){
    int t;
    cin>>t;
    while (t--){
        int n;
        cin>>n;
        int ar[n];
        for (int i = 0; i < n; ++i) {
            cin>>ar[i];
        }
        bool flag = false;
        for (int i = 0; i < n; ++i) {
            int x =floor(sqrt(ar[i]));
            if ( x*x != ar[i]){
                flag = true;
                break;
            }
        }
        if (flag){
            cout<<"YES\n";
        } else{
            cout<<"NO\n";
        }
    }
}
