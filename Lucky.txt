#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int num) {
    int num1=num%10;
    int num2=(num%100)/10;
    int num3=(num%1000)/100;
    int num4=(num%10000)/1000;
    int num5=(num%100000)/10000;
    int num6=(num%1000000)/100000;
    if((num1+num2+num3)==(num4+num5+num6)) {
        cout<<"YES"<<endl;
        return;
    }
    cout<<"NO"<<endl;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        findans(num);
    }
    return 0;
}
