#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int arr[],int num) {
    sort(arr,arr+num);
    if(num==1) {
        return 0;
    }
    int res=0;
    for(int i=0,j=num-1;i<(num/2);i++) {
        res=res+(arr[j]-arr[i]);
        j--;
    }
    return res;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        int arr[num];
        for(int i=0;i<num;i++) {
            cin>>arr[i];
        }
        cout<<findans(arr,num);
        cout<<endl;
    }
    return 0;
}
