#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int num) {
    num=num+1;
    for(int i=num;i<10000;i++) {
        int res=num;
        int n1=res%10;
        res=res/10;
        int n2=res%10;
        res=res/10;
        int n3=res%10;
        res=res/10;
        int n4=res%10;
        if(n1!=n2 && n1!=n3 && n1!=n4 && n2!=n3 && n2!=n4 && n3!=n4) {
            return num;
        }
        num++;
    }
    return 0;
}
int main()
{
    int num;
    cin>>num;
    cout<<findans(num);
    cout<<endl;
    return 0;
}
