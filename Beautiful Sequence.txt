#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int num,int arr[]) {
    for(int i=0;i<num;i++) {
        if(arr[i]<=(i+1)) {
            cout<<"YES"<<endl;
            return;
        }
    }
    cout<<"NO"<<endl;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        int arr[num];
        for(int i=0;i<num;i++) {
            cin>>arr[i];
        }
        findans(num,arr);
    }
    return 0;
}
