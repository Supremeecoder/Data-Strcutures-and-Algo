#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int gcd(int a, int b)
{
int res = min (a, b);
while (res>0)
{
if(a%res == 0 && b%res == 0)
{
  break;
}
res--;
}
return res;
}
string findans(int arr[],int num) {
    for(int j=0;j<num-1;j++) {
    for(int i=j+1;i<num;i++) {
        int ans=gcd(arr[j],arr[i]);
        if(ans<=2) {
            return "yes";
        }
    }
    }
    return "no";
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        int arr[num]; 
        for(int i=0;i<num;i++) {
            cin>>arr[i];
        }
        cout<<findans(arr,num);
        cout<<endl;
    }
    return 0;
}
