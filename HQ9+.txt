#include <iostream>
using namespace std;
void findans(string s) {
    for(int i=0;i<s.size();i++) {
        if(s[i]=='H'||s[i]=='Q'||s[i]=='9') {
            cout<<"YES"<<endl;
            return;
        }
    }
    cout<<"NO"<<endl;
}
int main()
{
    string s;
    cin>>s;
    findans(s);
    return 0;
}
