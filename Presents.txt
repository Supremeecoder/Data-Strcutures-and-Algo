#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int arr[],int num) {
    int arr1[num];
    for(int i=0;i<num;i++) {
        arr1[arr[i]-1]=i+1;
    }
    for(int i=0;i<num;i++) {
        cout<<arr1[i]<<" ";
    }
}
int main()
{
    int num;
    cin>>num;
    int arr[num];
    for(int i=0;i<num;i++) {
        cin>>arr[i];
    }
    findans(arr,num);
    return 0;
}
