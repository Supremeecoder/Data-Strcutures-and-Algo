#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(string s) {
    string s1="";
    s1=s1+s[0];
    for(int i=1;i<s.size()-1;i++) {
        if(s[i]==s[i+1]) {
            s1=s1+s[i];
            i++;
        }
    }
    s1=s1+s[s.size()-1];
    cout<<s1<<endl;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        string s;
        cin>>s;
        findans(s);
    }
    return 0;
}
