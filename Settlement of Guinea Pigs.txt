#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int num,int arr[]) {
    int flag=0;
    int i=0;
    for(;i<num;i++) {
        if(arr[i]==1) {
            flag=1;
            break;
        }
    }
    if(flag==0) {
        return 0;
    }
    int res=INT_MIN;
    int box=0;
    int fill=0;
    for(int j=i+1;j<num;j++) {
        if(arr[j]==1) {
            box++;
        } else {
            fill=box/2;
            box%=2;
        }
        res=max(box+fill,res);
    }
    return res+1;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        int arr[num];
        for(int i=0;i<num;i++) {
            cin>>arr[i];
        }
        cout<<findans(num,arr);
        cout<<endl;
    }
    return 0;
}
