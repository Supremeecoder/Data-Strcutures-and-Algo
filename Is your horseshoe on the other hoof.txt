#include <iostream>
using namespace std;
int findans(int arr[]) {
    int count=0;
    if(arr[0]==arr[1] || arr[0]==arr[2] || arr[0]==arr[3] || arr[0]==arr[4]) {
        count++;
    }
    if(arr[1]==arr[2] || arr[1]==arr[3] || arr[1]==arr[4]) {
        count++;
    }
    if(arr[2]==arr[3] || arr[2]==arr[4]) {
        count++;
    }
    if(arr[3]==arr[4]) {
        count++;
    }
    return count;
}
int main()
{
    int arr[4];
    for(int i=0;i<4;i++) {
        cin>>arr[i];
    }
    cout<<findans(arr);
    return 0;
}
