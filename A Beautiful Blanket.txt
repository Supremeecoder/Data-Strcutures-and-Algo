#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int row,int col) {
    int arr[row][col];
    int count=1;
    for(int i=1;i<=row;i++) {
        arr[i][1]=count++;
    }
    for(int i=1;i<=row;i++) {
        for(int j=2;j<=col;j++) {
            arr[i][j]=arr[i][j-1]+1024;
        }
    }
    cout<<row*col<<endl;
    for(int i=1;i<=row;i++) {
        for(int j=1;j<=col;j++) {
            cout<<arr[i][j]<<" ";
        }
        cout<<endl;
    }
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int row,col;
        cin>>row;
        cin>>col;
        findans(row,col);
    }
    return 0;
}
