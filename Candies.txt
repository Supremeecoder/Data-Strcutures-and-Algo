#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int num) {
    int x;
    for(int k=2;k<=35;k++) {
        int res=pow(2,k)-1;
        if(num%res) {
            continue;
        }
        x=num/res;
    }
    return x;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        cout<<findans(num);
        cout<<endl;
    }
    return 0;
}
