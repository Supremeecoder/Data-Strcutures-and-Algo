#include<iostream>
using namespace std;
int main()
{
    int num1,num2,min;
    cin>>num1;
    cin>>num2;
    if(num1>num2) {
        min=num2;
    } else {
        min=num1;
    }
    while(min>0) {
        if(num1%min==0 && num2%min==0) {
            break;
        } else {
            min--;
        }
    }
    cout<<min;
}
