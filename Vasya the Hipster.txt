#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int num1,int num2) {
    int res=min(num1,num2);
    int ans=(max(num1,num2)-res)/2;
    cout<<res<<" "<<ans;
}
int main()
{
    int num1;
    int num2;
    cin>>num1>>num2;
    findans(num1,num2);
    return 0;
}
