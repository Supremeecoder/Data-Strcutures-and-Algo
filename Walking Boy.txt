#include <iostream>
#include<bits/stdc++.h>
using namespace std;
string findans(int arr[],int num) {
    string s="no";
    int count=0;
    if(arr[0]>=120 && arr[0]<240) {
        count++;
    } else if(arr[0]>=240) {
        count=count+2;
    }
    for(int i=0;i<num-1;i++) {
        if(abs(arr[i]-arr[i+1])>=120) {
            count++;
            if(abs(arr[i]-arr[i+1])>=240) {
                count++;
                return "yes";
            }
        }
    }
    int res=1440-arr[num-1];
    if(res>=120 && res<240) {
        count++;
    } else if(res>=240) {
        count=count+2;
    }
    if(count>=2) {
            s="yes";
        }
    return s;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        int arr[num];
        for(int i=0;i<num;i++) {
            cin>>arr[i];
        }
        cout<<findans(arr,num);
        cout<<endl;
    }
    return 0;
}
