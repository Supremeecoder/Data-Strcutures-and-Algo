#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int num,int arr[]) {
    int countM=0;
    int countB=0;
    for(int i=0;i<num;i++) {
        if(arr[i]%2==0) {
            countM+=arr[i];
        } else {
            countB+=arr[i];
        }
    }
    if(countM>countB) {
        cout<<"YES"<<endl;
    } else {
        cout<<"NO"<<endl;
    }
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        int arr[num];
        for(int i=0;i<num;i++) {
            cin>>arr[i];
        }
        findans(num,arr);
    }
    return 0;
}
