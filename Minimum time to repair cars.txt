class Solution {
public:
    long long findans(long long mid,vector<int>& ranks,int &cars) {
        long long carsR=0;
        for(int i=0;i<ranks.size();i++) {
            carsR=carsR+(long long)sqrt(mid/ranks[i]);
        }
        if(carsR>=cars) {
            return 1;
        }
        return 0;
    }
    long long repairCars(vector<int>& ranks, int cars) {
        long long low=1;
        long long ans=0;
        long long high=100*(long long)cars*(long long)cars;
        while(low<=high) {
            long long mid=(high+low)/2;
            if(findans(mid,ranks,cars)) {
                high=mid-1;
                ans=mid;
            } else {
                low=mid+1;
            }
        }
        return ans;
    }
};
