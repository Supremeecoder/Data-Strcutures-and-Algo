class Solution {
public:
    int majorityElement(vector<int>& nums) {
        int len=nums.size();
        unordered_map<int,int> count;
        for(int i=0;i<len;i++) {
            count[nums[i]]++;
        }
        for(auto i:count) {
            if(i.second > (len/2)) {
                return i.first;
            }
        }
        return -1;
    }
};
