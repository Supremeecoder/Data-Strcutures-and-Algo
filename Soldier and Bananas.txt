#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int k,int n,int w) {
    int sum=0;
    for(int i=1;i<=w;i++) {
        sum=sum+i*k;
    }
    return n>sum?0:sum-n;
}
int main()
{
    int k;
    int n;
    int w;
    cin>>k>>n>>w;
    cout<<findans(k,n,w);
    return 0;
}
