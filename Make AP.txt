#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int a,int b,int c) {
    if((2*b-c)>0 && (2*b-c)%a==0) {
        cout<<"YES\n";
        return;
    } else if((a+c)%(2*b)==0) {
        cout<<"YES\n";
        return;
    } else if((2*b-a)>0 && (2*b-a)%c==0) {
        cout<<"YES\n";
        return;
    }
    cout<<"NO\n";
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int a,b,c;
        cin>>a>>b>>c;
        findans(a,b,c);
    }
    return 0;
}
