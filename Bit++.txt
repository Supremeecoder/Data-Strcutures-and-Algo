#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(string s) {
    for(int i=0;i<3;i++) {
        if(s[i]=='+') {
            return 1;
        } else if(s[i]=='-') {
            return -1;
        }
    }
    return 0;
}
int main()
{
        int num;
        cin>>num;
        int count=0;
        for(int i=0;i<num;i++) {
            string s;
            cin>>s;
            count=count+findans(s);
        }
        cout<<count;
    return 0;
}
