#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int n1,int n2,int n3) {
    if((n1+n2+n3)>=2) {
       return 1; 
    }
    return 0;
}
int main()
{
        int num1,num2,num3;
        int num;
        int count=0;
        cin>>num;
        for(int i=0;i<num;i++) {
            cin>>num1>>num2>>num3;
            count=count+findans(num1,num2,num3); 
        }
        cout<<count;
    return 0;
}
