#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(string s) {
    int countu=0;
    int countl=0;
    for(int i=0;i<s.size();i++) {
        if(isupper(s[i])) {
            countu++;
        } else {
            countl++;
        }
    }
    if(countl>=countu) {
        transform(s.begin(), s.end(), s.begin(), ::tolower);
    } else {
        transform(s.begin(), s.end(), s.begin(), ::toupper);
    }
    cout<<s;
}
int main()
{
    string s;
    cin>>s;
    findans(s);
    return 0;
}
