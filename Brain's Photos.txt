#include<iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int row,int col) {
    char arr[row][col];
    for(int i=0;i<row;i++) {
        for(int j=0;j<col;j++) {
            cin>>arr[i][j];
        }
    }
    for(int i=0;i<row;i++) {
        for(int j=0;j<col;j++) {
            if(arr[i][j]=='B'||arr[i][j]=='W'||arr[i][j]=='G') {
                continue;
            } else {
                cout<<"#Color"<<endl;
                return;
            }
        }
    }
    cout<<"#Black&White"<<endl;
}
int main()
{
    int row;
    int col;
    cin>>row>>col;
    findans(row,col);
    return 0;
}
