#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(string s) {
    int count=1;
    for(int i=0;i<s.size()-1;i++) {
        if(s[i]==s[i+1]) {
            count++;
            if(count>=7) {
                cout<<"YES";
                return;
            }
        } else {
            count=1;
        }
    }
    cout<<"NO";
}
int main()
{
    string s;
    cin>>s;
    findans(s);
    cout<<endl;
    return 0;
}
