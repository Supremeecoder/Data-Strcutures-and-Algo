#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int num) {
    int res=1;
    int count=0;
    while(res<num) {
        res=res*2;
        if(res>num) {
            res=res/2;
            break;
        }
        count++;
    }
    return pow(2,count)-1;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        cout<<findans(num);
        cout<<endl;
    }
    return 0;
}
