#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int num,int *arr) {
    int count=0;
    for(int i=1;i<=num;i++) {
        if(arr[i-1]==i) {
            count++;
        }
    }
    if(count%2==0) {
        return count/2;
    }
    return (count/2)+1;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        int arr[num];
        for(int i=0;i<num;i++) {
            cin>>arr[i];
        }
        cout<<findans(num,arr);
        cout<<endl;
    }
    return 0;
}
