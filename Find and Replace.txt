#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int num,string s) {
    int ans=0;
    unordered_map<char,int> mp;
    for(int i=0;i<num;i++) {
        if(mp.find(s[i])!=mp.end()) {
            if(mp[s[i]]!=i%2) {
                ans=1;
                break;
            }
        } else {
            mp[s[i]]=i%2;
        }
    }
    if(ans==0) {
        cout<<"YES"<<endl;
    } else {
        cout<<"NO"<<endl;
    }
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        string s;
        cin>>s;
        findans(num,s);
    }
    return 0;
}
