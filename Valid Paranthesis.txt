class Solution {
public:
    bool isValid(string s) {
        stack <char> s1;
        for(char x: s) {
            if(x=='{' || x=='[' || x=='(') {
                s1.push(x);
            } else {
                if(s1.empty()==true) {
                    return false;
                } else if(x==')' && s1.top()!='(') {
                    return false;
                } else if(x==']' && s1.top()!='[') {
                    return false;
                } else if(x=='}' && s1.top()!='{') {
                    return false;
                } else {
                    s1.pop();
                }
            }
        }
        return (s1.empty()==true);
    }
};
