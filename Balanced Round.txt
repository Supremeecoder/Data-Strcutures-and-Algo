#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int *arr,int num,int key) {
    if(num==1) {
        return 0;
    }
    int count=1;
    sort(arr,arr+num);
    int res=1;
    for(int i=1;i<num;i++) {
        if(abs(arr[i-1]-arr[i])<=key) {
            count++;
            res=max(res,count);
        } else {
            count=1;
        }
    }
    return num-res;
}
int main()
{
    int num;
    cin>>num;
    while(num--) {
        int num1;
        cin>>num1;
        int key;
        cin>>key;
        int arr[num1];
        for(int i=0;i<num1;i++) {
            cin>>arr[i];
        }
        cout<<findans(arr,num1,key);
        cout<<endl;
    }
    return 0;
}
