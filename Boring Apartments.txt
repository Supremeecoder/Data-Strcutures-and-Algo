#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int num) {
    int count=0;
    int ans=num%10;
    while(num>0) {
        int res=num%10;
        num=num/10;
        count++;
    }
    count=(count*(count+1))/2;
    return (ans-1)*10+count;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        cout<<findans(num);
        cout<<endl;
    }
    return 0;
}
