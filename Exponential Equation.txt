#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int num) {
    if(num%2==0) {
        cout<<"1"<<" "<<num/2<<endl;
    } else {
        cout<<"-1"<<endl;
    }
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        findans(num);
        cout<<endl;
    }
    return 0;
}
