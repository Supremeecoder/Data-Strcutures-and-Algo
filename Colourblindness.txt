#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(string s1,string s2) {
    if(s1.size()!=s2.size()) {
        cout<<"NO"<<endl;
        return;
    }
    for(int i=0;i<s1.size();i++) {
        if(s1[i]==s2[i]) {
            continue;
        } else if((s1[i]=='G' && s2[i]=='B') || (s1[i]=='B' && s2[i]=='G')) {
            continue;
        } else {
            cout<<"NO"<<endl;
            return;
        }
    }
    cout<<"YES"<<endl;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int size;

        findans(s1,s2);
    }

}
