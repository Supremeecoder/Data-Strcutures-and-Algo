#include <iostream>
#include<bits/stdc++.h>
using namespace std;
string findans(int num,string s) {
    string str="";
    if(num<4) {
      return "NO";
    }
    for(int i=0;i<num;i++) {
        if(s[i]=='m' || s[i]=='e' || s[i]=='o' || s[i]=='w' || s[i]=='M' || s[i]=='E' || s[i]=='O' || s[i]=='W') {
            continue;
        } else {
            return "NO";
        }
    }
    for(int i=0;i<num;i++) {
        if(tolower(s[i])!=tolower(s[i+1])) {
            str=str+s[i];
        }
    }
    transform(str.begin(), str.end(), str.begin(), ::tolower);
    if(str=="meow") {
      return "YES";
    }
    return "NO";
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        string s;
        cin>>s;
        cout<<findans(num,s);
        cout<<endl;
    }
    return 0;
}
