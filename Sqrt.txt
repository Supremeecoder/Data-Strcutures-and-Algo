class Solution {
public:
    int mySqrt(int x) {
        int low=0,high=x;
        int ans=-1;
        while(low<=high) {
            int mid=(high+low)/2;
            long mSq=(long)mid*(long)mid;
            if(mSq==x) {
                return mid;
            } else if(mSq>x) {
                high=mid-1;
            } else {
                low=mid+1;
                ans=mid;
            }
        }
        return ans;
    }
};
