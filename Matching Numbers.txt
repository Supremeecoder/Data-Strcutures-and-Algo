#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int num) {
    vector<pair<int,int>> v;
    if(num%2==0) {
        cout<<"no"<<endl;
        return;
    }
    cout<<"yes"<<endl;
    num=num*2;
    int j=num;
    for(int i=1;i<=num && i<j;i+=2) {
        v.push_back(make_pair(i,j));
        j--;
    }
    for(int i=2;i<=num && i<j;i+=2) {
        v.push_back(make_pair(i,j));
        j--;
    }
    for (int i=0; i<(num/2); i++)
    {
        cout << v[i].first << " "
             << v[i].second << endl;
    }
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        findans(num);
    }
    return 0;
}
