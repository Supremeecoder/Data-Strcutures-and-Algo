#include <iostream>
#include<bits/stdc++.h>
using namespace std;
string findans(string s,string t) {
    int i=0;
    int j=t.size()-1;
    if(s.size()!=t.size()) {
        return "NO";
    }
    while(i<s.size() && j>=0) {
        if(s[i]==t[j]) {
            i++;
            j--;
            continue;
        } else {
            return "NO";
        }
    }
    return "YES";
}
int main()
{
    string s;
    cin>>s;
    string t;
    cin>>t;
    cout<<findans(s,t);
    return 0;
}
