#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int arr[],int num) {
    int countpos=0;
    for(int i=0;i<num;i++) {
        if(arr[i]>=0) {
            countpos++;
        }
    }
    int rep=1;
    for(int i=1;i<=num;i++) {
        if(i<=countpos) {
            cout<<i<<" ";
        } else {
            cout<<countpos-rep<<" ";
            rep++;
        }
    }
    int res=1;
    int countneg=num-countpos;
    cout<<endl;
    for(int i=1;i<=num;i++) {
        if(countneg) {
            cout<<"1 "<<"0 ";
            countneg--;
            i++;
        } else {
            cout<<res<<" ";
            res++;
        }
    }
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        int arr[num];
        for(int i=0;i<num;i++) {
            cin>>arr[i];
        }
        findans(arr,num);
        cout<<endl;
    }
    return 0;
}
