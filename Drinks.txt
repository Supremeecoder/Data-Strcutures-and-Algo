#include <iostream>
#include<bits/stdc++.h>
using namespace std;
float findans(int arr[],int num) {
    float sum=0;
    float res=num;
    for(int i=0;i<num;i++) {
        sum=sum+arr[i];
    }
    return sum/res;
}
int main()
{

    int num;
    cin>>num;
    int arr[num];
    for(int i=0;i<num;i++) {
        cin>>arr[i];
    }
    cout<<findans(arr,num);
    return 0;
}
