#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(string s) {
    if(s.size()<=1) {
        cout<<s;
          return;
    }
    string s1="";
    sort(s.begin(),s.end());
    int i=0;
    for(;i<s.size();i++) {
        if(s[i]=='+') {
            continue;
        } else {
            break;
        }
    }
    for(int j=i;j<s.size()-1;j++) {
        s1=s1+s[j]+'+';
    }
    s1=s1+s[s.size()-1];
    cout<<s1;
}
int main()
{
    string s;
    cin>>s;
    findans(s);
    return 0;
}
