#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int arr[],int num) {
    int left[num]={0};
    int right[num]={0};
    for(int i=0;i<num;i++) {
        left[i]=arr[i];
        right[i]=arr[i];
    }
    for(int i=0;i<num-1;i++) {
        if(left[i]>=left[i+1]) {
            left[i+1]=left[i];
        }
    }
    if(left[num-2]>=left[num-1]) {
        left[num-1]=left[num-2];
    }
    for(int i=num-1;i>=1;i--) {
        if(right[i]>=right[i-1]) {
            right[i-1]=right[i];
        }
    }
    if(right[1]>=right[0]) {
        right[0]=right[1];
    }
    int ans=0;
    for(int i=0;i<num;i++) {
        ans=ans+min(left[i],right[i])-arr[i];
    }
    return ans;
}
int main()
{
    int num;
    cin>>num;
    int arr[num];
    for(int i=0;i<num;i++) {
        cin>>arr[i];
    }
    cout<<findans(arr,num);
    return 0;
}
