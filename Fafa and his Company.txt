#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int num) {
    int count=0;
    for(int i=1;i<=(num/2);i++) {
        if(num%i==0) {
            count++;
        }
    }
    return count;
}
int main()
{
    int num;
    cin>>num;
    cout<<findans(num);
    return 0;
}
