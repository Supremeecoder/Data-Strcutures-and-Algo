#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findK(int arr[],int num) {
    int count2=0;
    int ans=0;
    for(int i=0;i<num;i++) {
        if(arr[i]==2) {
            count2++;
        }
    }
    if(count2==0) {
        return 1;
    }
    if(count2%2==1) {
        return -1;
    }
    int res=count2/2;
    for(int i=0;i<num;i++) {
        if(arr[i]==2) {
            ans++;
        }
        if(ans==res) {
            break;
        }
    }
    return ans;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int n;
        cin>>n;
        int arr[n];
        for(int i=0;i<n;i++) {
            cin>>arr[i];
        }
        cout<<findK(arr,n)<<endl;;
    }
    return 0;
}
