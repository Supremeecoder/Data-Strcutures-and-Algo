#include <iostream>
#include<bits/stdc++.h>
using namespace std;

string findans(int arr[],int num) {
    for(int j=0;j<num-1;j++) {
    for(int i=j+1;i<num;i++) {
        int ans=__gcd(arr[j],arr[i]);
        if(ans<=2) {
            return "yes";
        }
    }
    }
    return "no";
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        int arr[num];
        for(int i=0;i<num;i++) {
            cin>>arr[i];
        }
        cout<<findans(arr,num);
        cout<<endl;
    }
    return 0;
}
