#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int num,string s) {
    int count[26]={0};
    for(int i=0;i<num;i++) {
        count[tolower(s[i])-'a']++;
    }
    for(int i=0;i<26;i++) {
        if(count[i]>=1) {
            continue;
        } else {
            cout<<"NO"<<endl;
            return;
        }
    }
    cout<<"YES"<<endl;
}
int main()
{
    int num;
    cin>>num;
    string s;
    cin>>s;
    findans(num,s);
    cout<<endl;
    return 0;
}
