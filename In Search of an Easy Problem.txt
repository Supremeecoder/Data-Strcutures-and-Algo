#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int arr[],int num) {
    for(int i=0;i<num;i++) {
      if(arr[i]==1) {
         cout<<"HARD";
         return;
      }
    }
    cout<<"EASY";
}
int main()
{
    int num;
    cin>>num;
    int arr[num];
    for(int i=0;i<num;i++) {
        cin>>arr[i];
    }
    findans(arr,num);
    return 0;
}
