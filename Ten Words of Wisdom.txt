#include <iostream>
using namespace std;
const int col=2;
int findans(int arr[][col],int num1) {
    int res=0;
    int ans=0;
    for(int i=0;i<num1;i++) {
        if(arr[i][0]<=10) {
            res=max(res,arr[i][1]);
            if(arr[i][1]>=res) {
                ans=i;
            }
        }
    }
    return ans+1;
}
int main()
{
    int num;
    cin>>num;
    while(num--) {
        int num1;
        cin>>num1;
        int res;
        int arr[num1][2];
        for(int i=0;i<num1;i++) {
            for(int j=0;j<2;j++) {
                cin>>arr[i][j];
            }
        }
        res=findans(arr,num1);
        cout<<res<<endl;
    }
    return 0;
}
