#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int num1,int num2) {
    if((num2-num1)>=2) {
        return 1;
    }
    return 0;
}
int main()
{
    int t;
    cin>>t;
    int count=0;
    while(t--) {
        int num1,num2;
        cin>>num1>>num2;
        count=count+findans(num1,num2);
    }
    cout<<count;
    return 0;
}
