#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int num) {
    int j=num;
    for(int i=0;i<num && j>=0;i++) {
            if((i+j)==num) {
                int sum1=0;
            int num1=i;
            int num2=j;
            while(num1>0) {
                int res=num1%10;
                sum1=sum1+res;
                num1=num1/10;
            }
            int sum2=0;
            while(num2>0) {
                int res=num2%10;
                sum2=sum2+res;
                num2=num2/10;
            }
            if(abs(sum1-sum2)==0 || abs(sum1-sum2)==1) {
            cout<<i<<" "<<j<<endl;
            break;
        }
            }
        j--;
    }
}
int main()
{
    int num;
    cin>>num;
    findans(num);
    return 0;
}
