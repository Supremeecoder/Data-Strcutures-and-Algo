#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(string s) {
    int count[26]={0};
    for(int i=0;i<s.size();i++) {
        if(isalpha(s[i])) {
            count[s[i]-'a']++;
        }
    }
    int ans=0;
    for(int i=0;i<26;i++) {
        if(count[i]>=1) {
            ans++;
        }
    }
    return ans;
}
int main()
{
    string s;
    getline(cin,s);
    cout<<findans(s);
    return 0;
}
