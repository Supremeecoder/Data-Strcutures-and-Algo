#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(string s1) {
    int count=0;
    for(int i=0;i<s1.size();i++) {
        if(s1[i]==s1[i+1]) {
            count++;
        } else {
            continue;
        }
    }
    return count;
}
int main()
{
    int num;
    string s1;
    cin>>num;
    cin>>s1;
    cout<<findans(s1);
    return 0;
}
