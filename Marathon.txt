#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int num1,int num2,int num3,int num4) {
    int count=0;
    if(num1<num2) {
        count++;
    }
    if(num1<num3) {
        count++;
    }
    if(num1<num4) {
        count++;
    }
    return count;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num1,num2,num3,num4;
        cin>>num1>>num2>>num3>>num4;
        cout<<findans(num1,num2,num3,num4);
        cout<<endl;
    }
    return 0;
}
