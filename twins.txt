#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int num,int arr[]) {
    int sum=0;
    sort(arr,arr+num);
    for(int i=0;i<num;i++) {
        sum=sum+arr[i];
    }
    int count=0;
    for(int i=num-1;i>=0;i--) {
        count=count+arr[i];
        if(count>(sum/2)) {
            return num-i;
        }
    }
    return 0;
}
int main()
{
    int num;
    cin>>num;
    int arr[num];
    for(int i=0;i<num;i++) {
        cin>>arr[i];
    }
    cout<<findans(num,arr);
    cout<<endl;
    return 0;
}
