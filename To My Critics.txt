#include <iostream>
using namespace std;
void findans(int num1,int num2,int num3) {
    if(num1+num2>=10) {
        cout<<"YES";
        return ;
    } else if(num1+num3>=10) {
        cout<<"YES";
        return ;
    } else if(num2+num3>=10) {
        cout<<"YES";
        return ;
    }
    cout<<"NO";
}
int main()
{
    int num;
    cin>>num;
    while(num--) {
        int n1,n2,n3;
        cin>>n1>>n2>>n3;
        findans(n1,n2,n3);
        cout<<endl;
    }
    return 0;
}
