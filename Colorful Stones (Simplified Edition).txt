#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(string s1,string s2) {
    int count=1;
    int i=0;
    int j=0;
    while(i<s1.size() && j<s2.size()) {
        if(s1[i]==s2[j]) {
            i++;
            count++;
        }
        j++;
    }
    return count;
}
int main()
{
    string s1,s2;
    cin>>s1>>s2;
    cout<<findans(s1,s2);
    cout<<endl;
    return 0;
}
