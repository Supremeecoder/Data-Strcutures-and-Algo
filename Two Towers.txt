#include <iostream>
#include<bits/stdc++.h>
using namespace std;
string findans(int len1,int len2,string s1,string s2) {
    string s3="yes";
    string s="";
    for(int i=s2.size()-1;i>=0;i--) {
        s=s+s2[i];
    }
    int count=0;
    string ans=s1+s;
    for(int i=0;i<ans.size()-1;i++) {
        if(ans[i]!=ans[i+1]) {
            continue;
        } else {
            count++;
            if(count>1) {
                s3="no";
                break;
            }
        }
    }
    return s3;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int len1;
        int len2;
        cin>>len1>>len2;
        string s1;
        string s2;
        cin>>s1>>s2;
        cout<<findans(len1,len2,s1,s2);
        cout<<endl;
    }
    return 0;
}
