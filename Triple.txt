#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int num,int arr[]) {
    sort(arr,arr+num);
    int count=0;
    for(int i=0;i<num-1;i++) {
        if(arr[i]==arr[i+1]) {
            count++;
            if(count==2) {
                return arr[i];
            }
        } else {
            count=0;
        }
    }
    return -1;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
    cin>>num;
    int arr[num];
    for(int i=0;i<num;i++) {
        cin>>arr[i];
    }
    cout<<findans(num,arr);
    cout<<endl;
    }
    return 0;
}
