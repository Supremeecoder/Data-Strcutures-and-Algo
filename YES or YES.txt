#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(string s) {
    if((s[0]=='y'||s[0]=='Y')&&(s[1]=='e'||s[1]=='E')&&(s[2]=='s'||s[2]=='S')) {
        cout<<"YES";
        return;
    }
    cout<<"NO";
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        string s;
        cin>>s;
        findans(s);
        cout<<endl;
    }
    return 0;
}
