#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int num1,int num2,int num3) {
    if((num1>num2 && num1<num3) || (num1>num3 && num1<num2)) {
        return num1;
    }
    if((num2>num1 && num2<num3) || (num2>num3 && num2<num1)){
        return num2;
    }
    return num3;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num1,num2,num3;
        cin>>num1>>num2>>num3;
        cout<<findans(num1,num2,num3);
        cout<<endl;
    }
    return 0;
}
