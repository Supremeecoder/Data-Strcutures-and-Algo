#include <iostream>
using namespace std;
int main()
{
    int num;
    cin>>num;
    while(num--) {
        char arr[8][8];
        string str="";
        for(int i=0;i<8;i++) {
            for(int j=0;j<8;j++) {
                cin>>arr[i][j];
            }
        }
        for(int i=0;i<8;i++) {
            for(int j=0;j<8;j++) {
                if(arr[i][j]=='.') {
                    continue;
                } else {
                    str=str+arr[i][j];
                }
            }
        }
        cout<<str<<endl;
    }
    return 0;
}
