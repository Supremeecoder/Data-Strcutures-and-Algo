#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int findans(int *arr,int num,int m,int k,int H) {
    int count=0;
    for(int i=0;i<num;i++) {
        if(abs(H-arr[i])%k==0 && (abs(H-arr[i])<=(m-1)*k)) {
            if((H-arr[i])==0) {
                continue;
            }
            count++;
        }
    }
    return count;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num,m,k,H;
        cin>>num>>m>>k>>H;
        int arr[num];
        for(int i=0;i<num;i++) {
            cin>>arr[i];
        }
        cout<<findans(arr,num,m,k,H);
        cout<<endl;
    }
    return 0;
}
