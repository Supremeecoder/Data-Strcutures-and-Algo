#include <bits/stdc++.h>
using namespace std;
int findans(int *arr,int num) {
    int res=INT_MAX;
    for(int i=0;i<num-1;i++) {
        if(arr[i]<=arr[i+1]) {
            continue;
        } else {
            return 0;
        }
    }
    for(int i=1;i<num;i++) {
        if(res>(arr[i]-arr[i-1])) {
            res=arr[i]-arr[i-1];
        }
    }
    return (res/2)+1;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        int arr[num];
        for(int i=0;i<num;i++) {
            cin>>arr[i];
        }
        cout<<findans(arr,num);
        cout<<endl;
    }
    return 0;
}
