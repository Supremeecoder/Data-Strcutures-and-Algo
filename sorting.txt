#include<iostream>
using namespace std;
void selectsort(int arr[],int num) {
    int temp[num];
    for(int i=0;i<num;i++) {
        int min_ind=0;
        for(int j=1;j<num;j++) {
            if(arr[j]<arr[min_ind]) {
                min_ind=j;
            }
        }
        temp[i]=arr[min_ind];
        arr[min_ind]=10000;
    }
    for(int i=0;i<num;i++) {
        arr[i]=temp[i];
    }
    for(int i=0;i<num;i++) {
        cout<<arr[i]<<" ";
    }
}
int main()
{
    int num;
    cin>>num;
    int arr[num];
    for(int i=0;i<num;i++) {
        cin>>arr[i];
    }
    for(int i=0;i<num;i++) {
        cout<<arr[i]<<" ";
    }
    cout<<endl;
    selectsort(arr,num);
    return 0;
}

#include<iostream>
using namespace std;
void selectsort(int arr[],int num) {
    for(int i=0;i<num-1;i++) {
        for(int j=0;j<num-i-1;j++) {
            if(arr[j]>arr[j+1]) {
                swap(arr[j],arr[j+1]);
            }
        }
    }
}
int main()
{
    int num;
    cin>>num;
    int arr[num];
    for(int i=0;i<num;i++) {
        cin>>arr[i];
    }
    for(int i=0;i<num;i++) {
        cout<<arr[i]<<" ";
    }
    cout<<endl;
    selectsort(arr,num);
    for(int i=0;i<num;i++) {
        cout<<arr[i]<<" ";
    }
    return 0;
}

                          INSERTSORT
#include<iostream>
using namespace std;
void insertsort(int arr[],int num) {
    for(int i=1;i<num;i++) {
        int key=arr[i];
        int j=i-1;
        while(j>=0 && arr[j]>key) {
            arr[j+1]=arr[j];
            j--;
        }
        arr[j+1]=key;
    }
}
int main()
{
    int num;
    cin>>num;
    int arr[num];
    for(int i=0;i<num;i++) {
        cin>>arr[i];
    }
    for(int i=0;i<num;i++) {
        cout<<arr[i]<<" ";
    }
    cout<<endl;
    insertsort(arr,num);
    for(int i=0;i<num;i++) {
        cout<<arr[i]<<" ";
    }
    return 0;
}
