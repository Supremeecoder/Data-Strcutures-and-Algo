Counting sort





#include <stdio.h>
#define n 20
#define k 10

int main()
{
    int pos[k], B[n];
    int A[] = {9, 8, 1, 2, 7, 6, 3, 4, 5, 6, 7, 2, 3, 4, 5, 9, 7, 4, 2, 5};
    int counter[k]; // here largest element is 9 k=9+1
    printf("unsortted array");
    for (int i = 0; i < n; i++){
        printf("%3d", A[i]);
    }
    for (int i = 0; i < k; i++){
        counter[i] = 0;
    }
    for (int i = 0; i < n; i++){
        counter[A[i]]++;
    }
    pos[0] = 0;
    for (int i = 1; i <= k; i++)
    {
        pos[i] = pos[i - 1] + counter[i - 1];
    }
    printf("\n\n");
    for(int i =0;i<n;i++){
        B[pos[A[i]]] = A[i];
        pos[A[i]] = pos[A[i]] + 1;
    }
    printf("\n sorted array after loop");
    for (int i = 0; i <= n; i++){
        printf("%4d", B[i]);
    }
    return 0;
}
