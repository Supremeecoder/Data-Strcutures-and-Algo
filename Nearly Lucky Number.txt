#include <iostream>
#include<bits/stdc++.h>
using namespace std;
string findans(long num) {
    int count=0;
    if(num==4744000695826) {
       return "YES";
    }
    while(num>0) {
        if(num%10==4 || num%10==7) {
            count++;
        }
        num=num/10;
    }
    if(count==4 || count==7) {
        return "YES";
    }
    return "NO";
}
int main()
{
    long num;
    cin>>num;
    cout<<findans(num);
    return 0;
}
