#include <iostream>
#include<bits/stdc++.h>
using namespace std;
void findans(int num,vector<int> v) {
    int res=0;
    for(auto& i:v) {
        res=res^i;
    }
    if(res==0) {
        cout<<"0";
        return;
    }
    if(num%2==1) {
        cout<<res;
    } else {
        cout<<"-1";
    }
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        int num;
        cin>>num;
        vector<int> v(num);
        for(auto& i:v) {
            cin>>i;
        }
        findans(num,v);
        cout<<endl;
    }
    return 0;
}
