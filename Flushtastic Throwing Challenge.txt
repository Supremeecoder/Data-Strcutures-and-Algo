#include <iostream>
using namespace std;
int main()
{
    int n;
    double c1,c2,r;
    int count=0;
    double x,y;
    cin>>n>>c1>>c2>>r;
    for(int i=0;i<n;i++) {
        cin>>x>>y;
        double boundary=(x-c1)*(x-c1)+(y-c2)*(y-c2)-r*r;
        if(boundary<=0) {
            count++;
        }
    }
    cout<<count<<endl;
    return 0;
}
